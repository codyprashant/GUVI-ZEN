Applying array functions on Rest countries data

Solving problems using array functions on restcountries data.
Get all the countries from Asia continent using Filter function

                      
    var request = new XMLHttpRequest()
    var url_string = 'https://restcountries.eu/rest/v2/all';
    
    // Open a new connection, using the GET request on the URL endpoint
    request.open('GET',url_string , true) 
    request.send();
    request.onload = function() {
        var data = JSON.parse(this.response);
        var filteredData = data.filter((singleData) => singleData.region == "Asia")
        console.log(filteredData);
    }

Print the total population of countries using reduce function

                          
        var request = new XMLHttpRequest()
        var url_string = 'https://restcountries.eu/rest/v2/all';
        
        // Open a new connection, using the GET request on the URL endpoint
        request.open('GET',url_string , true) 
        request.send();
        request.onload = function() {
            var data = JSON.parse(this.response);
            var filteredData = data.reduce((total, singleData) => {return singleData.population + parseInt(total)}, 0)
            console.log(filteredData);
        }

Print the total population of countries in Asia continent using reduce and filter function

                          
        var request = new XMLHttpRequest()
        var url_string = 'https://restcountries.eu/rest/v2/all';
        
        // Open a new connection, using the GET request on the URL endpoint
        request.open('GET',url_string , true) 
        request.send();
        request.onload = function() {
            var data = JSON.parse(this.response);
            var filteredData = data.filter((singleData) => singleData.region == "Asia").reduce((total, singleData) => {
              return singleData.population + parseInt(total)
            }, 0)
            console.log(filteredData);
        }
